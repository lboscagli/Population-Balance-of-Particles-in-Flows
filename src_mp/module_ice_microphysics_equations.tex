\subsection*{Depositional Ice Growth (Bier et al. 2021)}
From `pbe_depositional_growth_ice_Bier`:
\begin{align}
&r_{part} = (3/4\pi\,v)^{1/3},\\
&S_v = \begin{cases}S_{eq,water}(r_{part},T)+1 & v_m>v_{0,max}\\ S_{eq}(r_{part},r_{nuc},T,\kappa)+1 & \text{otherwise}\end{cases},\\
&\lambda_v = 2 D_{v,c}(T,p)\sqrt{\frac{1}{2R_v T}},
\quad \beta_v = \frac{r_{part}}{r_{part}+\lambda_v}+\frac{4 D_{v,c}(T,p)}{0.5 r_{part} v_{th}(T)},\\
&\lambda_a = 0.8\frac{k_{air}(T)T}{p}\sqrt{\frac{1}{2R_a T}},
\quad \beta_k = \frac{r_{part}}{r_{part}+\lambda_a}+\frac{4 k_{air}(T)}{0.7 C_p v_{th}(T) \rho_{air,dry}},\\
&\text{denom}\_{dm/dt} = \left( \frac{D_{v,c}(T,p)\beta_v^{-1}L_d S_v p_{sat}^{ice}}{k_{air}(T)\beta_k^{-1}\beta_v^{-1}T}\right)
\left(\frac{L_d}{R_v T}-1\right) + R_v T,\\
&\frac{dm}{dt} = 4\pi r_{part}
D_{v,c}(T,p)\beta_v^{-1}\frac{S^*p_{sat}^{liq}-S_v p_{sat}^{ice}}{\text{denom}\_{dm/dt}},\\
&\frac{dr}{dt} = \frac{1}{4\pi r_{part}^2 \rho_{ice}}\frac{dm}{dt},\\
&g_{coeff,1,r} = \begin{cases}4\pi (3/4\pi)^{2/3} dr/dt & dr/dt\ge0\\0&\text{else}\end{cases},
\\
&g_{coeff,2} = \frac{2}{3}.
\end{align}

\subsection*{Freezing Temperature Criterion}
From `pbe_freezing_temperature`:
\begin{align}
&\text{LWV} =
\begin{cases}v_m - \frac{4\pi}{3}r_{vc}^3 & \text{if } active\le1\text{ or }v_m=v_{0,min}\\ v_m & \text{else}\end{cases},\\
&J_{freeze} = 10^{6} e^{a_1 T + a_2},\\
&\text{jet cooling rate }\Lambda = -\beta \frac{T_{0j}-T_{amb}}{\tau_m} \left(\frac{T-T_{amb}}{T_{0j}-T_{amb}}\right)^{1+1/\beta},\\
&\tau_m = \frac{r_{0j}\sqrt{2/\epsilon_t}}{u_{0j}},\\
&T_{frz} = \begin{cases}0 & \text{if } LWV=0\\ \frac{1}{a_1}\left[\ln\left(\frac{10^{-6}a_1 \Lambda}{LWV}\right)-a_2\right]&\text{otherwise}\end{cases}.
\end{align}
(Here $a_1=-3.5714$, $a_2=858.719$, $\epsilon_t=0.0285$, $\beta=0.9$,
$r_{0j}=0.5 d_{jet}$.)
